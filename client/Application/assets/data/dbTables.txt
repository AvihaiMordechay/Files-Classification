user:
- id: INTEGER PRIMARY KEY NOT NULL (from firebase uid)
- name; STRING NOT NULL
- email: STRING NOT NULL

folders:
- id: INTEGER PRIMARY KEY NOT NULL AUTOINCREMENT
- name: STRING PRIMARY KEY NOT NULL
- filesCount: INTEGER NOT NULL
 
files:
- id INTEGER PRIMARY KEY NOT NULL AUTOINCREMENT
- name: STRING PRIMARY KEY NOT NULL
- folderId: STRING FOREIGN KEY NOT NULL (to folders/name)
- tagName: STRING NOT NULL
- type: STRING NOT NULL
- path: STRING NOT NULL

favorites:
- id INTEGER PRIMARY KEY AUTOINCREMENT,
- name: STRING PRIMARY KEY FOREIGN KEY NOT NULL (to files->name)
